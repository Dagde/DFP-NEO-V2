    });

  // in each return path separately to ensure they're properly initialized

  console.warn('‚ö†Ô∏è initializeData reached end without returning data, returning empty data');
  return {
    instructors: [],
    trainees: [],
    events: [],
    scores: new Map(),
    pt051Assessments,
    courses: [],
    courseColors: {},
    archivedCourses: {},
    coursePriorities: [],

  };
}
