model Personnel {
  id             String   @id @default(cuid())
  userId         String
  name           String
  rank           String?
  role           String
  qualifications Json?
  availability   Json?
  preferences    Json?
  isActive       Boolean  @default(true)
  createdAt      DateTime @default(now())
  updatedAt      DateTime @updatedAt
  user           User     @relation("Personnel", fields: [userId], references: [id], onDelete: Cascade)

  @@index([userId])
  @@index([role])
  @@index([isActive])
}
