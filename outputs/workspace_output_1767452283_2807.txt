'use client';

import { useState } from 'react';
import { signIn } from 'next-auth/react';
import { useRouter } from 'next/navigation';
import Image from 'next/image';

export default function LoginPage() {
  const router = useRouter();
  const [username, setUsername] = useState('');
  const [password, setPassword] = useState('');
  const [error, setError] = useState('');
  const [isLoading, setIsLoading] = useState(false);

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    setError('');
    setIsLoading(true);

    try {
      const result = await signIn('credentials', {
        username,
        password,
        redirect: false,
      });

      if (result?.error) {
        setError('Invalid username or password');
        setIsLoading(false);
        return;
      }

      router.push('/select');
    } catch (error) {
      setError('An error occurred. Please try again.');
      setIsLoading(false);
    }
  };

  return (
    <div className="min-h-screen w-full bg-black flex flex-col items-center justify-center p-4">
      <div className="w-full flex flex-col items-center justify-center" style={{ height: '100vh', gap: '2rem' }}>
        {/* Logo Section with Reflection - Smaller to fit screen */}
        <div className="relative flex-shrink-0 w-full max-w-4xl px-4">
          <Image
            src="/images/logo.png"
            alt="DFP-NEO Logo"
            width={900}
            height={300}
            unoptimized
            className="w-full h-auto"
          />
          {/* Reflection Effect */}
          <div className="absolute top-full left-0 right-0 opacity-20 pointer-events-none overflow-hidden" style={{ height: '80px' }}>
            <Image
              src="/images/logo.png"
              alt=""
              width={900}
              height={300}
              unoptimized
              className="w-full h-auto transform scale-y-[-1]"
              style={{
                maskImage: 'linear-gradient(to bottom, rgba(0,0,0,0.3) 0%, transparent 50%)',
                WebkitMaskImage: 'linear-gradient(to bottom, rgba(0,0,0,0.3) 0%, transparent 50%)'
              }}
            />
          </div>
        </div>

        {/* Login Form with Metal Plate Background */}
        <form onSubmit={handleSubmit} className="w-full space-y-4 flex-shrink-0">
          {/* Metal Plate Container - Small and visible */}
          <div className="relative w-full max-w-lg mx-auto" style={{ minHeight: '200px' }}>
            {/* Metal Plate Background - Make sure it's visible */}
            <div className="absolute inset-0 z-0">
              <Image
                src="/images/metal-plate-input.png"
                alt=""
                width={500}
                height={150}
                unoptimized
                className="w-full h-auto"
              />
            </div>
            
            {/* Input Fields Container - Positioned on top */}
            <div className="relative z-10 px-8 py-6 space-y-3" style={{ minHeight: '200px' }}>
              {/* Username Row */}
              <div className="flex items-center gap-3">
                <div className="w-28 flex-shrink-0">
                  <Image
                    src="/images/username.png"
                    alt="Username"
                    width={300}
                    height={60}
                    unoptimized
                    className="w-full"
                  />
                </div>
                <input
                  type="text"
                  value={username}
                  onChange={(e) => setUsername(e.target.value)}
                  className="flex-1 bg-transparent border-2 border-gray-600/40 rounded-lg px-3 py-2 text-gray-200 text-sm focus:outline-none focus:border-gray-500/60 transition-colors placeholder:text-gray-500"
                  placeholder="Joe.Bloggs"
                  required
                  disabled={isLoading}
                />
              </div>

              {/* Password Row */}
              <div className="flex items-center gap-3">
                <div className="w-28 flex-shrink-0">
                  <Image
                    src="/images/password.png"
                    alt="Password"
                    width={300}
                    height={60}
                    unoptimized
                    className="w-full"
                  />
                </div>
                <input
                  type="password"
                  value={password}
                  onChange={(e) => setPassword(e.target.value)}
                  className="flex-1 bg-transparent border-2 border-gray-600/40 rounded-lg px-3 py-2 text-gray-200 text-sm focus:outline-none focus:border-gray-500/60 transition-colors placeholder:text-gray-500"
                  placeholder="**********"
                  required
                  disabled={isLoading}
                />
              </div>
            </div>
          </div>

            {/* Input Fields Container - Positioned on top of metal plate */}
            <div className="absolute inset-0 flex flex-col justify-center px-8 py-6 space-y-4">
              {/* Username Row */}
              <div className="flex items-center gap-4">
                <div className="w-32 flex-shrink-0">
                  <Image
                    src="/images/username.png"
                    alt="Username"
                    width={400}
                    height={80}
                    unoptimized
                    className="w-full"
                  />
                </div>
                <input
                  type="text"
                  value={username}
                  onChange={(e) => setUsername(e.target.value)}
                  className="flex-1 bg-transparent border-2 border-gray-600/40 rounded-lg px-4 py-2 text-gray-200 text-sm focus:outline-none focus:border-gray-500/60 transition-colors placeholder:text-gray-500"
                  placeholder="Joe.Bloggs"
                  required
                  disabled={isLoading}
                />
              </div>

              {/* Password Row */}
              <div className="flex items-center gap-4">
                <div className="w-32 flex-shrink-0">
                  <Image
                    src="/images/password.png"
                    alt="Password"
                    width={400}
                    height={80}
                    unoptimized
                    className="w-full"
                  />
                </div>
                <input
                  type="password"
                  value={password}
                  onChange={(e) => setPassword(e.target.value)}
                  className="flex-1 bg-transparent border-2 border-gray-600/40 rounded-lg px-4 py-2 text-gray-200 text-sm focus:outline-none focus:border-gray-500/60 transition-colors placeholder:text-gray-500"
                  placeholder="**********"
                  required
                  disabled={isLoading}
                />
              </div>
            </div>
          </div>

          {/* Error Message */}
          {error && (
            <div className="bg-red-900/30 border border-red-500 text-red-200 px-6 py-3 rounded-lg text-sm text-center max-w-xl mx-auto">
              {error}
            </div>
          )}

          {/* Submit Button */}
          <div className="flex justify-center pt-4">
            <button
              type="submit"
              disabled={isLoading}
              className="bg-gray-800/80 hover:bg-gray-700/80 border-2 border-gray-700/50 text-gray-200 px-16 py-4 rounded-lg text-lg font-semibold tracking-wider uppercase transition-all disabled:opacity-50 disabled:cursor-not-allowed shadow-lg hover:shadow-xl"
            >
              {isLoading ? 'Authenticating...' : 'Access System'}
            </button>
          </div>
        </form>
      </div>
    </div>
  );
}
