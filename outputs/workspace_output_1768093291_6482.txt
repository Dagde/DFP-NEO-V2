export async function fetchInstructors(): Promise<any[]> {
  const result = await fetchAPI<{ personnel: any[] }>('/personnel?role=INSTRUCTOR');
  if (result.success && result.data?.personnel) {
    return result.data.personnel;
  }
  return [];
}
