    // Fix solo flight display - ensure SOLO flights show correctly
    (function() {
      function fixSoloFlights() {
        // Find all flight tiles/events
        const flightElements = document.querySelectorAll('[class*="flight"], [class*="event"], [data-event-type]');
        
        flightElements.forEach(element => {
          const text = element.textContent;
          
          // Check if this is a solo flight (SCT, SOLO, or similar indicators)
          const isSolo = text.includes('SCT') || 
                        text.includes('SOLO') || 
                        text.includes('[SOLO]') ||
                        text.includes('Solo');
          
          if (isSolo) {
            // Find duplicate names in the element
            const nameMatches = text.match(/([A-Z][a-z]+,\s*[A-Z][a-z]+)/g);
            
            if (nameMatches &amp;&amp; nameMatches.length >= 2) {
              // Check if the same name appears twice (indicating incorrect solo display)
              const firstName = nameMatches[0];
              const secondName = nameMatches[1];
              
              if (firstName === secondName) {
                // This is a solo flight showing the same pilot twice
                // Find the second occurrence and replace with "SOLO"
                const innerHTML = element.innerHTML;
                let count = 0;
                const fixedHTML = innerHTML.replace(new RegExp(firstName.replace(/[.*+?^${}()|[\]\\]/g, '\\$&amp;'), 'g'), (match) => {
                  count++;
                  return count === 1 ? match : '<strong style="color: #fbbf24;">SOLO</strong>';
                });
                
                if (fixedHTML !== innerHTML) {
                  element.innerHTML = fixedHTML;
                  console.log('Fixed solo flight display for:', firstName);
                }
              }
            }
          }
        });
      }
      
      // Run immediately
      fixSoloFlights();
      
      // Run periodically
      setInterval(fixSoloFlights, 2000);
      
      // Run on DOM changes
      if (typeof MutationObserver !== 'undefined') {
        const observer = new MutationObserver(fixSoloFlights);
        observer.observe(document.body, {
          childList: true,
          subtree: true
        });
      }
